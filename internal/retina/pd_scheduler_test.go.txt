func TestNewProbingDirectiveScheduler(t *testing.T) {
	cooldown := 5 * time.Second
	scheduler := NewProbingDirectiveScheduler(cooldown)

	if scheduler == nil {
		t.Fatal("NewProbingDirectiveScheduler returned nil")
	}
	if scheduler.directives == nil {
		t.Error("directives slice is nil")
	}
	if len(scheduler.directives) != 0 {
		t.Errorf("expected empty directives, got %d", len(scheduler.directives))
	}
	if scheduler.cooldowns == nil {
		t.Error("cooldowns slice is nil")
	}
	if scheduler.cooldownPeriod != cooldown {
		t.Errorf("expected cooldown %v, got %v", cooldown, scheduler.cooldownPeriod)
	}
}

func TestPDScheduler_Set_Shuffles(t *testing.T) {
	scheduler := NewProbingDirectiveScheduler(time.Second)

	// Create many directives to make shuffle observable
	directives := make([]*api.ProbingDirective, 100)
	for i := 0; i < 100; i++ {
		directives[i] = &api.ProbingDirective{AgentID: string(rune('a' + i))}
	}

	_ = scheduler.Set(directives)

	// Check that original slice is not modified
	sameOrder := true
	for i := 0; i < len(directives); i++ {
		if scheduler.directives[i] != directives[i] {
			sameOrder = false
			break
		}
	}

	// With 100 elements, probability of same order is essentially 0
	if sameOrder {
		t.Error("directives were not shuffled")
	}
}

func TestPDScheduler_Set_DoesNotModifyOriginal(t *testing.T) {
	scheduler := NewProbingDirectiveScheduler(time.Second)

	directives := []*api.ProbingDirective{
		{AgentID: "1"},
		{AgentID: "2"},
		{AgentID: "3"},
	}

	original := make([]*api.ProbingDirective, len(directives))
	copy(original, directives)

	_ = scheduler.Set(directives)

	for i, d := range directives {
		if d.AgentID != original[i].AgentID {
			t.Error("Set modified original slice")
		}
	}
}

func TestPDScheduler_Select_ContextCancelled(t *testing.T) {
	scheduler := NewProbingDirectiveScheduler(time.Hour) // Long cooldown

	directives := []*api.ProbingDirective{{AgentID: "1"}}
	_ = scheduler.Set(directives)

	// First select should work
	_, _ = scheduler.Select(context.Background())

	// Set cooldown to future
	scheduler.mutex.Lock()
	scheduler.cooldowns[0] = time.Now().Add(time.Hour)
	scheduler.mutex.Unlock()

	ctx, cancel := context.WithCancel(context.Background())
	cancel() // Cancel immediately

	_, err := scheduler.Select(ctx)
	if err != context.Canceled {
		t.Errorf("expected context.Canceled, got %v", err)
	}
}

	scheduler.mutex.Lock()
	scheduler.cooldowns[0] = time.Now().Add(time.Hour)
	scheduler.mutex.Unlock()

	ctx, cancel := context.WithTimeout(context.Background(), 10*time.Millisecond)
	defer cancel()

	_, err := scheduler.Select(ctx)
	if err != context.DeadlineExceeded {
		t.Errorf("expected context.DeadlineExceeded, got %v", err)
	}
}

func TestPDScheduler_CooldownInitialization(t *testing.T) {
	scheduler := NewProbingDirectiveScheduler(time.Second)

	before := time.Now()
	directives := []*api.ProbingDirective{{AgentID: "1"}, {Id: "2"}}
	_ = scheduler.Set(directives)
	after := time.Now()

	for i, cooldown := range scheduler.cooldowns {
		if cooldown.Before(before) || cooldown.After(after) {
			t.Errorf("cooldown[%d] not set to current time", i)
		}
	}
}

func TestPDScheduler_Select_ImmediateWhenNotOnCooldown(t *testing.T) {
	scheduler := NewProbingDirectiveScheduler(time.Hour)

	directives := []*api.ProbingDirective{{AgentID: "1"}}
	_ = scheduler.Set(directives)

	// Set cooldown to past
	scheduler.mutex.Lock()
	scheduler.cooldowns[0] = time.Now().Add(-time.Hour)
	scheduler.mutex.Unlock()

	start := time.Now()
	_, err := scheduler.Select(context.Background())
	elapsed := time.Since(start)

	if err != nil {
		t.Errorf("Select failed: %v", err)
	}
	if elapsed > 50*time.Millisecond {
		t.Errorf("Select took too long: %v", elapsed)
	}
}

func TestPDScheduler_Select_AfterSet(t *testing.T) {
	scheduler := NewProbingDirectiveScheduler(time.Millisecond)

	// Set directives
	directives := []*api.ProbingDirective{
		{AgentID: "1"},
		{AgentID: "2"},
		{AgentID: "3"},
	}
	_ = scheduler.Set(directives)

	// All cooldowns should be in the past now (or very close)
	time.Sleep(2 * time.Millisecond)

	selected, err := scheduler.Select(context.Background())
	if err != nil {
		t.Errorf("Select failed: %v", err)
	}

	// Verify selected is one of our directives
	found := false
	for _, d := range directives {
		if selected == d {
			found = true
			break
		}
	}
	if !found {
		t.Error("selected directive not in original set")
	}
}
